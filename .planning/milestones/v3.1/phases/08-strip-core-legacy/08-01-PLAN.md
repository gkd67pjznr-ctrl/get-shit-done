Rewrite to:

```javascript
function findPhaseInternal(cwd, phase, milestoneScope) {
  if (!phase) return null;

  const normalized = normalizePhaseName(phase);

  // When milestoneScope is provided, search milestone workspace phases
  if (milestoneScope) {
    const root = planningRoot(cwd, milestoneScope);
    const phasesDir = path.join(root, 'phases');
    const relBase = path.join('.planning', 'milestones', milestoneScope, 'phases');
    return searchPhaseInDir(phasesDir, relBase, normalized);
  }

  // Auto-detect milestone by searching all milestone phase dirs
  const milestonesDir = path.join(cwd, '.planning', 'milestones');
  try {
    const msDirs = fs.readdirSync(milestonesDir, { withFileTypes: true })
      .filter(e => e.isDirectory())
      .map(e => e.name)
      .sort()
      .reverse(); // newest first

    for (const ms of msDirs) {
      const phasesDir = path.join(milestonesDir, ms, 'phases');
      if (!fs.existsSync(phasesDir)) continue;
      const relBase = path.join('.planning', 'milestones', ms, 'phases');
      const result = searchPhaseInDir(phasesDir, relBase, normalized);
      if (result) {
        result.milestone_scope = ms;
        return result;
      }
    }
  } catch {}
  return null;
}
```

- Remove `const layout = detectLayoutStyle(cwd);` and the `if (layout === 'milestone-scoped')` guard
- The milestone auto-detect body is now unconditional (it was previously inside the if block)
- DELETE the entire legacy branch (lines 284-313): `// Legacy layout: search current phases first` through the archived milestone search
- The function now always searches milestone dirs when no milestoneScope is explicitly provided
