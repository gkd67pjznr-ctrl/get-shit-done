```js
const STOP_WORDS = new Set([
  'a', 'an', 'the', 'and', 'or', 'but', 'in', 'on', 'at', 'to', 'for',
  'of', 'with', 'is', 'are', 'was', 'be', 'it', 'this', 'that', 'as',
  'by', 'from', 'not', 'use', 'used', 'when', 'any', 'all', 'also',
]);

function tokenize(text) {
  return new Set(
    text.toLowerCase()
      .split(/[\s\-_.,;:!?()\[\]{}'"\/\\]+/)
      .filter(t => t.length > 1 && !STOP_WORDS.has(t))
  );
}

function jaccardScore(setA, setB) {
  if (setA.size === 0 && setB.size === 0) return 0;
  const intersection = [...setA].filter(t => setB.has(t)).length;
  const union = new Set([...setA, ...setB]).size;
  return union === 0 ? 0 : intersection / union;
}
```

**`parseDecisions(content)`** function:

Accepts the full text of PROJECT.md as a string. Returns an array of `{ decision, rationale }` objects parsed from the Key Decisions table.

Algorithm:
1. Find the `## Key Decisions` section heading in `content`.
2. From that position, extract lines until the next `##` heading or end of string.
3. For each line that matches the pattern `| ... | ... | ... |` (3+ pipe-separated columns), skip header and separator lines (lines containing only `|`, `-`, and spaces).
4. For each data row, split by `|`. The trimmed second column is `decision`, the trimmed third column is `rationale`. Ignore the fourth column (Outcome).
5. Skip rows where `decision` or `rationale` is empty or is a header value (`Decision`, `Rationale`).
6. Return the array. If no rows are found or no `## Key Decisions` section exists, return [].

```js
function parseDecisions(content) {
  const decisions = [];
  const sectionMatch = content.match(/## Key Decisions[\s\S]*?(?=\n## |\n$|$)/);
  if (!sectionMatch) return decisions;
  const section = sectionMatch[0];
  const lines = section.split('\n');
  for (const line of lines) {
    if (!line.startsWith('|')) continue;
    const cols = line.split('|').map(c => c.trim()).filter((_, i) => i > 0);
    if (cols.length < 3) continue;
    const [decision, rationale] = cols;
    if (!decision || !rationale) continue;
    if (decision === 'Decision' || decision.replace(/-/g, '').trim() === '') continue;
    decisions.push({ decision, rationale });
  }
  return decisions;
}
```

**`matchCorrectionsToDecision(corrections, decision)`** function:

Accepts an array of correction objects and a `{ decision, rationale }` object. Returns an array of `{ correction, score }` pairs where `score >= MIN_JACCARD`.

Algorithm:
1. Build `decisionTokens = tokenize(decision.decision + ' ' + decision.rationale)`.
2. For each correction in `corrections`:
   - Build `correctionTokens = tokenize((correction.correction_to || '') + ' ' + (correction.diagnosis_text || ''))`.
   - Compute `score = jaccardScore(decisionTokens, correctionTokens)`.
   - If `score >= MIN_JACCARD`, include `{ correction, score }` in results.
3. Return the results array.

```js
function matchCorrectionsToDecision(corrections, decision) {
  const decisionTokens = tokenize(decision.decision + ' ' + decision.rationale);
  const matches = [];
  for (const correction of corrections) {
    const corrText = (correction.correction_to || '') + ' ' + (correction.diagnosis_text || '');
    const corrTokens = tokenize(corrText);
    const score = jaccardScore(decisionTokens, corrTokens);
    if (score >= MIN_JACCARD) {
      matches.push({ correction, score });
    }
  }
  return matches;
}
```

**`detectTensions(options)`** function:

Accepts `{ cwd? }`. Returns array of tension objects. Never throws.

```js
function detectTensions(options) {
  try {
    const cwd = (options && options.cwd) ? options.cwd : process.cwd();

    // Read PROJECT.md
    const projectPath = path.join(cwd, '.planning', 'PROJECT.md');
    if (!fs.existsSync(projectPath)) return [];
    const content = fs.readFileSync(projectPath, 'utf-8');

    // Parse decisions
    const decisions = parseDecisions(content);
    if (decisions.length === 0) return [];

    // Read active corrections
    const { readCorrections } = require('./write-correction.cjs');
    const corrections = readCorrections({ status: 'active' }, { cwd });
    if (corrections.length === 0) return [];

    // Match corrections to each decision
    const tensions = [];
    for (const decision of decisions) {
      const matches = matchCorrectionsToDecision(corrections, decision);
      if (matches.length < MIN_TENSION_CORRECTIONS) continue;
      const avgScore = matches.reduce((sum, m) => sum + m.score, 0) / matches.length;
      tensions.push({
        decision_text: decision.decision,
        rationale: decision.rationale,
        matched_corrections: matches.map(m => m.correction),
        confidence: avgScore,
        correction_count: matches.length,
      });
    }

    return tensions;
  } catch (e) {
    return [];
  }
}
```

```js
function makeProjectMd(rows) {
  // rows: array of { decision, rationale } objects
  const rowLines = rows.map(r => `| ${r.decision} | ${r.rationale} | ✓ Good |`).join('\n');
  return `# Project\n\n## Key Decisions\n\n| Decision | Rationale | Outcome |\n|----------|-----------|---------||\n${rowLines}\n\n## Other Section\n`;
}

function makeCorrection(overrides = {}) {
  return {
    id: overrides.id || 'corr-001',
    diagnosis_category: overrides.diagnosis_category || 'code.wrong_pattern',
    status: overrides.status || 'active',
    timestamp: new Date().toISOString(),
    correction_to: overrides.correction_to || 'fix it',
    diagnosis_text: overrides.diagnosis_text || '',
    scope: 'project',
    correction_from: 'old',
    phase: '80',
    session_id: 'sess-001',
    source: 'self_report',
  };
}

function createFixture(corrections = [], projectMdContent = '') {
  const tmp = fs.mkdtempSync(path.join(os.tmpdir(), 'gsd-da-'));
  const patternsDir = path.join(tmp, '.planning', 'patterns');
  fs.mkdirSync(patternsDir, { recursive: true });
  fs.writeFileSync(path.join(tmp, '.planning', 'config.json'), JSON.stringify({}));

  if (projectMdContent) {
    fs.writeFileSync(path.join(tmp, '.planning', 'PROJECT.md'), projectMdContent);
  }
  return tmp;
}
```
